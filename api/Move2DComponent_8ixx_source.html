184 /**
185 * @brief Sets the movement direction and throttle.
186 *
187 * @details Updates the target rotation angle based on the input direction
188 * and calculates the shortest angular distance to the target.
189 *
190 * @param direction Normalized 2D direction vector from analog stick.
191 * @param throttle Magnitude of the stick input (0.0 to 1.0).
192 */
193 void move(helios::math::vec2f direction, float throttle) {
194 if (throttle <= helios::math::EPSILON_LENGTH) {
195 steeringInput_ = helios::math::vec2f{0.0f, 0.0f};
196 throttle_ = 0.0f;
197 isInputActive_ = false;
198 return;
199 }
200
201 steeringInput_ = direction;
202 throttle_ = throttle;
203
204 assert(std::abs(direction.length() - 1.0f) <= 0.001f && "Unexpected direction vector - not normalized");
205
206
207 targetRotationAngle_ = helios::math::degrees(std::atan2(direction[1], direction[0]));
208 rotationAngleDelta_ = std::fmod((targetRotationAngle_ - currentRotationAngle_) + 540.0f, 360.0f) - 180.0f;
209
210
211 isInputActive_ = true;
212
213 float turnBoost = 1.0f + 0.5f*std::clamp((abs(rotationAngleDelta_))/180.f, 0.0f, 1.0f);
214 currentRotationSpeed_ = turnBoost * rotationSpeed_ * throttle_;
215 currentMovementSpeed_ = movementSpeed_ * throttle_;
216
217 }
