24 /**
25 * @brief Input component for twin-stick gamepad control schemes.
26 *
27 * @details Translates gamepad analog stick input into movement and aiming commands.
28 * The left stick controls movement direction and speed, while the right stick
29 * controls aiming direction and intensity.
30 *
31 * Each frame, this component reads the current input snapshot and generates
32 * Move2DCommand and Aim2DCommand instances that are queued in the CommandBuffer
33 * for later execution.
34 *
35 * @note Requires the owning GameObject to have Move2DComponent and Aim2DComponent
36 * attached for the generated commands to have any effect.
37 */
38 class TwinStickInputComponent : public helios::engine::game::Updatable, public helios::engine::game::Component {
39
40 /**
41 * @brief Flag to indicate whether shoot commands should be derived
42 * from the aim component.

44 * If true, ShootCommands will be created from dedicated input.
45 */
46 bool useDedicatedShootInput_ = false;

50 /**
51 * @brief Processes gamepad input and generates movement/aiming commands.
52 *
53 * @param updateContext Context containing input snapshot and command buffer.
54 */
55 void update(helios::engine::game::UpdateContext& updateContext) noexcept override {
56
57 auto* inputSnapshot = updateContext.inputSnapshot();
58 auto* commandBuffer = updateContext.commandBuffer();
59
60 // Left stick: movement
61 const auto leftStick = inputSnapshot->gamepadState().left();
62 float finalSpeed = 0.0f;
63 float speed = leftStick.length();
64 auto ldir = helios::math::vec2f{0.0f, 0.0f};
65
66 if (speed > helios::math::EPSILON_LENGTH) {
67 ldir = leftStick.normalize();
68 finalSpeed = speed;
69 }
70
71 commandBuffer->add(
72 gameObject()->guid(),
73 std::make_unique<helios::engine::game::commands::Move2DCommand>(ldir, finalSpeed)
74 );
75
76 // Right stick: aiming
77 const auto rightStick = inputSnapshot->gamepadState().right();
78 float freq = rightStick.length();
79 float finalFreq = 0.0f;
80 auto rdir = helios::math::vec2f{0.0f, 0.0f};
81
82 if (freq > helios::math::EPSILON_LENGTH) {
83 rdir = rightStick.normalize();
84 finalFreq = freq;
85 }
86
87 commandBuffer->add(
88 gameObject()->guid(),
89 std::make_unique<helios::engine::game::commands::Aim2DCommand>(rdir, finalFreq)
90 );
91
92 if (useDedicatedShootInput_) {
93 // right trigger: shooting
94 const auto rightTrigger = inputSnapshot->gamepadState().triggerRight();
95
96 commandBuffer->add(
97 gameObject()->guid(),
98 std::make_unique<helios::engine::game::commands::ShootCommand>(rightTrigger)
99 );
100 } else {
101 commandBuffer->add(
102 gameObject()->guid(),
103 std::make_unique<helios::engine::game::commands::ShootCommand>(finalFreq)
104 );
105 }
106
107
108
109 }
