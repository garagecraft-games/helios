101 *
102 * @details
103 * Calculates the corrected position and reflected velocity when an entity
104 * collides with level boundaries. The restitution coefficient determines
105 * how much velocity is preserved after the bounce.
106 *
107 * @param worldTranslation Current world position of the entity.
108 * @param objectBounds World-space AABB of the entity.
109 * @param levelBounds World-space AABB of the level.
110 * @param m2d Reference to the Move2DComponent for velocity modification.
111 * @param bbc Reference to the LevelBoundsBehaviorComponent for restitution.
112 *
113 * @return Corrected world position after bounce.
114 */
115 helios::math::vec3f bounce(
116 helios::math::vec3f worldTranslation,
117 helios::math::aabbf objectBounds,
118 helios::math::aabbf levelBounds,
119 helios::engine::game::components::physics::Move2DComponent& m2d,
120 helios::engine::game::components::physics::LevelBoundsBehaviorComponent& bbc
121 ){
122
123
124 auto velocity = m2d.velocity();
125 bool hitX = false;
126 bool hitY = false;
127 auto translation = worldTranslation;
128
129 // Check left boundary collision
130 if (objectBounds.min()[0] < levelBounds.min()[0]) {
131 translation[0] += (levelBounds.min()[0] - objectBounds.min()[0]);
132 if (velocity[0] < 0) {
133 velocity = (velocity.withX(bbc.restitution() * -velocity[0]));
134 }
135 hitX = true;
136 // Check right boundary collision
137 } else if (objectBounds.max()[0] > levelBounds.max()[0]) {
138 translation[0] -= (objectBounds.max()[0] - levelBounds.max()[0]) ;
139 if (velocity[0] > 0) {
140 velocity = (velocity.withX(bbc.restitution() * -velocity[0]));
141 }
142 hitX = true;
143 }
144
145 objectBounds = objectBounds + (translation - worldTranslation);
146
147 // Check bottom boundary collision
148 if (objectBounds.min()[1] < levelBounds.min()[1]) {
149 translation[1] += (levelBounds.min()[1] - objectBounds.min()[1]);
150 if (velocity[1] < 0) {
151 velocity = velocity.withY(bbc.restitution() * -velocity[1]);
152 }
153 hitY = true;
154 // Check top boundary collision
155 } else if (objectBounds.max()[1] > levelBounds.max()[1]) {
156 translation[1] -= (objectBounds.max()[1] - levelBounds.max()[1]);
157 if (velocity[1] > 0) {
158 velocity = velocity.withY(bbc.restitution() * -velocity[1]);
159 }
160 hitY = true;
161 }
162
163 if (hitX || hitY) {
164 m2d.setVelocity(velocity);
165 }
