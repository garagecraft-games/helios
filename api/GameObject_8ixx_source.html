20 /**
21 * @brief Base class for interactive game entities using a component-based architecture.
22 *
23 * @details A GameObject is a container for Component instances that define its behavior.
24 * Each GameObject has a unique Guid for identification in the GameWorld and supports
25 * per-frame updates for components implementing the Updatable interface.
26 *
27 * GameObjects are typically owned by a GameWorld and should be created using
28 * std::make_unique before being added to the world.
29 *
30 * Example usage:
31 * ```cpp
32 * auto entity = std::make_unique<helios::engine::game::GameObject>();
33 * entity->add<SceneNodeComponent>(sceneNode);
34 * entity->add<Move2DComponent>();
35 *
36 * auto* ptr = world.addGameObject(std::move(entity));
37 * ```
38 */
39 class GameObject {
40
41 protected:
42 /**
43 * @brief Unique identifier for this GameObject.
44 *
45 * @details Generated during construction and remains constant for the lifetime
46 * of the object. Used for lookups in GameWorld and command targeting.
47 */
48 helios::util::Guid guid_;
49
50 /**
51 * @brief List of components attached to this GameObject.
52 *
53 * @details Components are owned by the GameObject via std::unique_ptr.
54 */
55 std::vector<std::unique_ptr<Component>> components_;
56
57 /**
58 * @brief Cached list of components that implement the Updatable interface.
59 *

61 * The pointers are non-owning references to elements within components_.

70 /**
71 * @brief Adds a new component of type T to the GameObject.
72 *
73 * @tparam T The type of component to add. Must derive from Component.
74 * @tparam Args Argument types for the component's constructor.
75 * @param args Arguments forwarded to the component's constructor.
76 * @return Reference to the newly created component.
77 *
78 * @details The component is created, attached to this GameObject (onAttach is called),
79 * and if it implements Updatable, it is added to the update list.
80 */
81 template<typename T, typename... Args>
82 T& add(Args&&... args) {
83
84 auto component_ptr = std::make_unique<T>(std::forward<Args>(args)...);
85 T* raw_component_ptr = component_ptr.get();
86
87 if constexpr (std::derived_from<T, helios::engine::game::Updatable>) {
88 updatables_.push_back(raw_component_ptr);
89 }
90 components_.push_back(std::move(component_ptr));
91
92 // make sure component is registered before onAttach is called,
93 // in case onAttach implementations query the gameObject for **this** component
94 // Use raw_component_ptr instead of component_ptr since ownership was moved above
95 raw_component_ptr->onAttach(this);
96
97 return *raw_component_ptr;
98 }
99
100 /**
101 * @brief Retrieves an existing component of type T, or creates and adds it if not present.
102 *
103 * @tparam T The type of component to retrieve or add. Must derive from Component.
104 * @tparam Args Argument types for the component's constructor.
105 * @param args Arguments forwarded to the component's constructor if it needs to be created.
106 * @return Reference to the existing or newly created component.
107 *
108 * @details This is a convenience method that combines get() and add().
109 * If the component already exists, the arguments are ignored.
110 */
111 template<typename T, typename... Args>
112 T& getOrAdd(Args&&... args) {
113
114 if (auto* cmp = get<T>()) {
115 return *cmp;
116 }
117
118 return add<T>(std::forward<Args>(args)...);
119 }
120
121 /**
122 * @brief Retrieves a component of type T.
123 *
124 * @tparam T The type of component to retrieve.
125 * @return Pointer to the component if found, nullptr otherwise.
126 *
127 * @details Performs a linear search through the components.
128 */
129 template<typename T>
130 T* get() const {
131 for (const auto& component : components_) {
132 if (auto* comp = dynamic_cast<T*>(component.get())) {
133 return comp;
134 }
135 }
136
137 return nullptr;
138 }
139
140 /**
141 * @brief Checks if the GameObject has a component of type T.
142 *
143 * @tparam T The type of component to check for.
144 * @return true if the component exists, false otherwise.
145 */
146 template<typename T>
147 [[nodiscard]] bool has() const {
148 return get<T>() != nullptr;
149 }

151 /**
152 * @brief Updates all updatable components attached to this GameObject.
153 *
154 * @param updateContext Context containing frame delta time and other update data.
155 */
156 void update(helios::engine::game::UpdateContext& updateContext) {
