129 /**
130 * @brief Sets the shooting intensity.
131 *
132 * @details Call with a non-zero value to start shooting, or zero to stop.
133 * The intensity affects whether projectiles are spawned during update().
134 *
135 * @param intensity Fire intensity (0.0 to 1.0). Zero stops firing.
136 * @param sourceVelocity The velocity of the object emitting the projectile.
137 */
138 void shoot(float intensity, helios::math::vec3f sourceVelocity) {
139
140 if (intensity == 0.0f || !aimComponent_) {
141 intensity_ = 0.0f;
142 return;
143 }
144 sourceVelocity_ = sourceVelocity;
145 intensity_ = intensity;
146 }
147
148 /**
149 * @brief Updates the component, spawning projectiles when ready.
150 *
151 * @details Checks the cooldown timer and aim direction. If conditions are met,
152 * spawns a projectile from the ProjectilePoolSystem at the current position traveling in
153 * the aim direction.
154 *
155 * @param updateContext The update context containing timing and game state.
156 */
157 void update(helios::engine::game::UpdateContext& updateContext) noexcept override {
158
159 cooldownDelta_ += updateContext.deltaTime() * intensity_;
160
161 if (intensity_ == 0.0f) {
162 return;
163 }
164
165 const auto aimDir = aimComponent_->direction();
166
167 if (cooldownDelta_ >= cooldownTime_ && aimDir.length() > helios::math::EPSILON_LENGTH) {
168 cooldownDelta_ = 0;
169 auto* projectilePoolSystem = updateContext.gameWorld()
170 ->get<helios::engine::game::systems::gameplay::ProjectilePoolSystem>();
171
172 auto aimDirNorm = aimDir.normalize().toVec3();
173
174 projectilePoolSystem->spawn(
175 transformComponent_->localTranslation(),
176 aimDirNorm * projectileSpeed_, aimDirNorm, sourceVelocity_
177 );
178 }
179 }
