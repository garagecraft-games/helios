99 * @brief Computes rotation for a GameObject based on its Move2DComponent state.
100 *
101 * @details Calculates the rotation matrix based on current rotation angle, applying
102 * dampening when input is inactive. Rotation speed decreases exponentially
103 * until the threshold is reached.
104 *
105 * @param gameObject Pointer to the GameObject to rotate.
106 * @param deltaTime Frame delta time in seconds.

110 [[nodiscard]] helios::math::mat4f rotateGameObject(helios::engine::game::GameObject* gameObject, float deltaTime) noexcept {
111 auto* cmp = gameObject->get<helios::engine::game::components::physics::Move2DComponent>();
112
113 assert(cmp != nullptr && "Unexpected invalid gameObject passed");
114 assert(deltaTime >= 0 && "Unexpected negative value for deltaTime");
115
116 if (deltaTime == 0) {
117 return helios::math::mat4f::identity();
118 }
119
120 float currentRotationSpeed = cmp->currentRotationSpeed();
121 float currentRotationAngle = cmp->currentRotationAngle();
122 float rotationAngleDelta = cmp->rotationAngleDelta();
123 bool isInputActive = cmp->isInputActive();
124
125 // Apply rotation dampening when input is inactive
126 if (!isInputActive) {
127 currentRotationSpeed *= std::pow(cmp->rotationDampening(), deltaTime);
128
129 if (currentRotationSpeed < cmp->rotationSpeedThreshold()) {
130 currentRotationSpeed = 0.0f;
131 }
132
133 cmp->setCurrentRotationSpeed(currentRotationSpeed);
134 }
135
136 // Apply rotation step towards target angle
137 if (currentRotationSpeed > 0.0f) {
138 float rotationStep = std::copysign(
139 std::min(
140 std::abs(rotationAngleDelta),
141 currentRotationSpeed * deltaTime
142 ),
143 rotationAngleDelta
144 );
145 currentRotationAngle = currentRotationAngle + rotationStep;
146 rotationAngleDelta = std::fmod((cmp->targetRotationAngle() - currentRotationAngle) + 540.0f, 360.0f) - 180.0f;
147
148 // Snap to target when close enough
149 if (std::abs(rotationAngleDelta) <= 0.5f) {
150 currentRotationAngle = cmp->targetRotationAngle();
151 rotationAngleDelta = 0.0f;
152 currentRotationSpeed = 0.0f;
153 }
154
155 cmp->setCurrentRotationAngle(currentRotationAngle);
156 cmp->setRotationAngleDelta(rotationAngleDelta);
157 cmp->setCurrentRotationSpeed(currentRotationSpeed);
158 }
159
160 return helios::math::rotate(
161 helios::math::mat4f::identity(),
162 helios::math::radians(currentRotationAngle),
163 helios::math::vec3f(0.0f, 0.0f, 1.0f)
164 );
165
166 };

168 /**
169 * @brief Computes translation delta for a GameObject based on its Move2DComponent state.
170 *
171 * @details Calculates velocity changes based on input state. When input is active,
172 * accelerates in the facing direction. When inactive, applies exponential
173 * drag to slow down. Velocity is clamped to maximum speed.
174 *
175 * @param gameObject Pointer to the GameObject to move.
176 * @param deltaTime Frame delta time in seconds.
177 *
178 * @return Translation delta to apply to the entity this frame.
179 */
180 [[nodiscard]] helios::math::vec3f moveGameObject(helios::engine::game::GameObject* gameObject, float deltaTime) {
181
182 auto* cmp = gameObject->get<helios::engine::game::components::physics::Move2DComponent>();
183
184 assert(cmp != nullptr && "Unexpected invalid gameObject passed");
185 assert(deltaTime >= 0 && "Unexpected negative value for deltaTime");
186
187 if (deltaTime == 0) {
188 return helios::math::vec3f{0.0f};
189 }
190
191 float currentRotationAngle = cmp->currentRotationAngle();
192 bool isInputActive = cmp->isInputActive();
193
194
195 auto velocity = cmp->velocity();
196
197 if (!isInputActive) {
198 float movementDampening = cmp->movementDampening();
199
200 // Apply exponential drag when no input is active.
201 // This creates a smooth deceleration effect (velocity approaches zero over time).
202 const float drag = std::pow(movementDampening, deltaTime);
203 velocity = velocity * drag;
204
205 if (drag <= helios::math::EPSILON_LENGTH) {
206 velocity = helios::math::vec3f{0.0f, 0.0f, 0.0f};
207 }
208
209 } else {
210 // Compute the current facing direction from the actual rotation angle.
211 // This converts the angle to a unit vector in the XY plane.
212 const auto currentDirection = helios::math::vec3f(
213 cos(helios::math::radians(currentRotationAngle)), sin(helios::math::radians(currentRotationAngle)), 0.0f
214 );
215
216 // Accelerate in the current facing direction.
217 // Uses throttle (input intensity) to scale acceleration.
218 velocity = velocity + currentDirection * (cmp->movementAcceleration() * cmp->throttle() * deltaTime);
219 }
220
221 // Clamp velocity to maximum speed to prevent unlimited acceleration.
222 if (velocity.length() > cmp->movementSpeed()) {
223 velocity = velocity.normalize() * cmp->movementSpeed();
224 }
225
226 cmp->setVelocity(velocity);
227
228 return velocity * deltaTime;
229 }
