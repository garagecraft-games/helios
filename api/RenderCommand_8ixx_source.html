16 /**
17 * @brief DTO for storing rendering-specific command information to be
18 * passed to the RenderQueue.
19 *
20 * A `RenderCommand` does not extend the lifetime of the `shader_` and the `mesh_`,
21 * as they are both referenced by weak pointers. If the referenced shader or the mesh
22 * are no longer valid when this command is about to be executed,
23 * this `RenderCommand` should be ignored for rendering.
24 */
25 class RenderCommand {
26 /**
27 * @brief A non-owning, weak reference to the shader program to be used.
28 */

31 /**
32 * @brief A non-owning, weak reference to the mesh this command uses.
33 */

35
36 /**
37 * @brief An owning, unique pointer to the uniform values specific for the object to be rendered.

40 */

43 /**
44 * @brief An owning, unique pointer to the uniform values specific to the material of the object to
