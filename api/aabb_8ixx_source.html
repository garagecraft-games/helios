18 /**
19 * @brief Axis-Aligned Bounding Box for spatial culling and collision detection.
20 *
21 * An AABB is defined by two corner points: a minimum and maximum point in 3D space.
22 * Depending on the use case, the box is aligned to the world or local coordinate axes.
23 *
24 * Common use cases include frustum culling, broad-phase collision detection,
25 * and spatial partitioning in scene graphs.
26 *
27 * @tparam T The numeric type of the vector components (typically `float` or `double`).
28 *
29 * @see [Gla95, pp. 548-550] Glassner, A. (1995). Graphics Gems
30 * @see [DP11, pp. 304-311] Dunn, F., & Parberry, I. (2011). 3D Math Primer for Graphics and Game Development
31 */
32 template<helios::math::Numeric T>
33 struct aabb {

52 /**
53 * @brief Constructs an empty AABB with inverted min/max values.
54 *
55 * The AABB is initialized with min set to the maximum representable value
56 * and max set to the minimum representable value. This allows the first
57 * `add()` call to properly establish the initial bounds.

126 /**
127 * @brief Checks if this AABB fully contains another AABB.
128 *
129 * @details Tests whether all corners of the specified bounding box lie within
130 * the bounds of this AABB. Both minimum and maximum corners must be contained.
131 *
132 * @param box The AABB to test for containment.
133 *
134 * @return True if the specified box is fully contained within this AABB, false otherwise.
135 */
136 [[nodiscard]] constexpr bool contains(const helios::math::aabb<T>& box) const noexcept {
137
138 auto v_min = box.min();
139 auto v_max = box.max();
140
141 return v_min[0] >= min_[0] && v_min[1] >= min_[1] && v_min[2] >= min_[2] &&
142 v_max[0] <= max_[0] && v_max[1] <= max_[1] && v_max[2] <= max_[2];
143
144 }

146 /**
147 * @brief Translates the AABB by a given vector.
148 *
149 * Creates a new AABB by adding the components of the given translation vector

158 min_ + v, max_ + v

160 }

164 /**
165 * @brief Expands the AABB to include a given point.
166 *
167 * Updates the minimum and maximum bounds to ensure the specified point
168 * lies within the AABB. Each component is compared independently:
169 * - If a component of the point is smaller than the current minimum, the minimum is updated.
170 * - If a component of the point is larger than the current maximum, the maximum is updated.
171 *
172 * @param point The 3D point to include in the bounding box.
173 */
174 void add(const helios::math::vec3<T>& point) noexcept {
175 min_[0] = std::min(min_[0], point[0]);
176 min_[1] = std::min(min_[1], point[1]);
177 min_[2] = std::min(min_[2], point[2]);
178
179 max_[0] = std::max(max_[0], point[0]);
180 max_[1] = std::max(max_[1], point[1]);
181 max_[2] = std::max(max_[2], point[2]);
182 }

184 /**
185 * @brief Transforms the AABB by a 4x4 transformation matrix.
186 *
187 * Applies a transformation matrix to the AABB and returns a new axis-aligned bounding box
188 * that fully contains the transformed original. This method preserves the axis-aligned property
189 * by computing new min/max bounds from the transformed corners.
190 *
191 * The algorithm efficiently transforms only the min/max corners instead of all 8 vertices,
192 * using Arvo's optimization technique from Graphics Gems.
193 *
194 * @param mat The 4x4 transformation matrix to apply.
195 *
196 * @return A new AABB that fully contains the transformed bounding box.
197 *
198 * @see [Gla95, pp. 548-550] "Transforming Axis-Aligned Bounding Boxes" by James Arvo
199 */
200 [[nodiscard]] aabb<T> applyTransform(const mat4<T>& mat) const noexcept {
201
202 const T translationX = static_cast<T>(mat(0, 3));
203 const T translationY = static_cast<T>(mat(1, 3));
204 const T translationZ = static_cast<T>(mat(2, 3));
205
206 vec3<T> newMin = {translationX, translationY, translationZ};
207 vec3<T> newMax = newMin;
208
209 for (int i = 0; i < 3; ++i) {
210 for (int j = 0; j < 3; ++j) {
211
212 T val = static_cast<T>(mat(j, i));
213
214 T e = val * min_[i];
215 T f = val * max_[i];
216
217 if (e < f) {
218 newMin[j] += e;
219 newMax[j] += f;
220 } else {
221 newMin[j] += f;
222 newMax[j] += e;
223 }
224 }
225 }
226
227 return aabb<T>(newMin, newMax);
228 }

257 * @tparam T The numeric type of the vector components.
