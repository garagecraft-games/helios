108 /**

110 *
111 * This method takes a value in the specified unit and converts it to helios units,
112 * e.g. spatial units to METER, temporal values to SECONDS.
113 *
114 * @param v Value to convert.
115 * @param unit Unit of the value to convert.
116 *
117 * @return Value expressed in temporal or spatial helios units.
118 */
119 [[nodiscard]] constexpr float from(const float v, const Unit unit) noexcept {
120 switch (unit) {
121 case Unit::Meter:
122 return fromM(v);
123 case Unit::Centimeter:
124 return fromCm(v);
125 case Unit::Seconds:
126 return fromS(v);
127 case Unit::MilliSeconds:
128 return fromMs(v);
129 default:
130 std::unreachable();
131 }
