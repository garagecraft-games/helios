15 /**
16 * @class FpsMetrics
17 * @brief Aggregates and analyzes frame timing data over a rolling window.
18 *
19 * FpsMetrics collects FrameStats data from multiple frames and calculates
20 * average FPS and frame times. It maintains a configurable history buffer
21 * for smoothing out momentary fluctuations in frame rate.
22 *
23 * This class is particularly useful for debugging, profiling, and displaying
24 * performance metrics in development tools or debug overlays.
25 *
26 * @par Usage Example:
27 * ```cpp
28 * helios::tooling::FpsMetrics metrics;
29 * metrics.setHistorySize(120);
30 *
31 * // In game loop:
32 * helios::engine::FrameStats stats = framePacer.sync();
33 * metrics.addFrame(stats);
34 *
35 * float fps = metrics.getFps();
36 * float avgFrameTime = metrics.getFrameTimeMs();
37 * ```
38 */
39 class FpsMetrics {
40
41 /**
42 * @brief Stores a rolling history of frame statistics.
43 *
44 * `history_` holds the full timing data (work time, wait time, total frame time)
45 * for a specified number of recent frames. This buffer is used to compute
46 * average frame rate and frame time metrics over a configurable window.
47 *
48 * The use of a `std::deque` allows efficient addition and removal
49 * of frame timing records while maintaining the order of events.
50 */
51 std::deque<helios::engine::FrameStats> history_;
52
53 /**
54 * @brief Size of the history buffer for storing frame statistics.
55 *
56 * Determines the maximum number of frames for which the frame timing data
57 * is retained to calculate performance metrics such as average FPS and
58 * frame times. A larger history size smooths out short-term variations
59 * but may reduce responsiveness to sudden changes in performance.
60 *
61 * This variable is initialized to a default value of 60 but can be adjusted
62 * based on the application's requirements for performance analysis.
63 */
64 size_t historySize_ = 60;
65
66 /**
67 * @brief Stores the calculated average frames per second (FPS).
68 *
69 * The `avgFps_` variable holds the computed average FPS value based on the
70 * accumulated frame timing data in `history_`. It is recomputed lazily when
71 * one of the query functions (e.g. `getFps()`) is called after new frames
72 * have been added.
73 */
74 float avgFps_ = 0.0f;
75
76 /**
77 * @brief Stores the average frame time in milliseconds.
78 *
79 * Represents the total frame time (including work and wait) averaged over
80 * the frames currently stored in the history buffer. The value is stored
81 * in milliseconds and recomputed lazily when queried via the public getters
82 * after new frames have been added.
83 */
84 float avgFrameTime_ = 0.0f;
85
86 /**
87 * @brief Tracks the most recent frame's work time in milliseconds.
88 *
89 * Represents the duration of work performed during the last processed frame.
90 * This value is derived from the most recent `FrameStats` in the history and
91 * expressed in milliseconds. It is updated when the metrics are recomputed
92 * (lazy evaluation triggered by the getters).
93 */
94 float lastWorkTime_ = 0.0f;
95
96 /**
97 * @brief Tracks the idle time in milliseconds for the most recent frame.
98 *
99 * Represents the duration the system waited (idle time) during the last
100 * processed frame. This value is derived from the most recent `FrameStats`
101 * in the history and expressed in milliseconds. It is updated when the
102 * metrics are recomputed (lazy evaluation triggered by the getters).
103 */
104 float lastWaitTime_ = 0.0f;
105
106 /**
107 * @brief Tracks the total number of frames processed since initialization.
108 *
109 * The frameCount_ variable accumulates the count of frames that have been
110 * added via `addFrame()`. It starts at 0 and increments with each call,
111 * serving as a global counter independent of the current history size.
112 */
113 unsigned long long frameCount_ = 0;
114
115 /**
116 * @brief Indicates whether cached metrics need to be recomputed.
117 *
118 * Set to true whenever new frame data is added or the history changes in a
119 * way that invalidates the cached values. The next call to one of the public
120 * query functions will trigger a recomputation in `update()`.
121 */
122 bool needsUpdate_ = true;
123
124 /**
125 * @brief Recomputes cached metrics if needed.
126 *
127 * If `needsUpdate_` is true and the history is non-empty, this function
128 * updates `lastWorkTime_`, `lastWaitTime_`, `avgFrameTime_` and `avgFps_`
129 * based on the current contents of `history_` and then clears the flag.
130 *
131 * If the history is empty or the cache is already up to date, this function
132 * returns immediately without modifying any values.
133 *
134 * @note This function is safe to call in hot paths and does not throw under
135 * normal conditions; it is therefore marked `noexcept`.
136 */
137 void update() noexcept;
138

141 /**
142 * @brief Adds a frame's statistics to the metrics history.
143 *
144 * Processes the provided FrameStats, updates the rolling history,
145 * and marks cached values as dirty so they will be recomputed on
146 * the next query. Old frames are automatically removed when the
147 * history buffer exceeds the configured size. Additionally, the
148 * `frameCount_` member is increased by one for each call.
149 *
150 * @param stats Frame statistics to add to the history.
151 *
152 * @note When used together with FramePacer, a typical usage pattern is:
153 * @code
154 * helios::engine::FrameStats stats = framePacer.sync();
155 * metrics.addFrame(stats);
156 * @endcode
157 * This should usually be called once per frame at the end of the
158 * frame loop, after timing information for the current frame has
159 * been measured.
160 */
161 void addFrame(const helios::engine::FrameStats& stats);
162
163 /**
164 * @brief Sets the size of the frame history buffer.
165 *
166 * Determines how many frames are used for calculating averages.
167 * Larger values produce smoother metrics but respond slower to changes.
168 *
169 * @param size Number of frames to keep in history (default: 60).
170 *
171 * @note Common values: 30 for roughly half-second average at 60 FPS,
172 * 60 for roughly one-second average at 60 FPS.
173 */
