17 /**
18 * @brief Abstract interface for processing input and generating game commands.
19 *
20 * @details The `InputHandler` translates raw input data, captured in an `InputSnapshot`,
21 * into executable `Commands`. Handlers receive an immutable input snapshot and generate
22 * commands that are queued in a `CommandBuffer` for later execution.
23 *
24 * Example implementation:
25 * ```cpp
26 * class PlayerInputHandler : public helios::engine::game::InputHandler {
27 * public:
28 * void handleInput(
29 * const InputSnapshot& snapshot,
30 * const GameObject& target,
31 * CommandBuffer& buffer,
32 * float deltaTime
33 * ) override {
34 * auto& pad = snapshot.gamepadState();
35 *
36 * // Movement
37 * if (pad.left().length() > 0.1f) {
38 * helios::math::vec3f dir(pad.left()[0], 0.0f, -pad.left()[1]);
39 * buffer.add(target, std::make_unique<MoveCommand>(dir, deltaTime * 5.0f));
40 * }
41 *
42 * // Jump
43 * if (pad.buttonA()) {
44 * buffer.add(target, std::make_unique<JumpCommand>());
45 * }
46 * }
47 * };
48 * ```
