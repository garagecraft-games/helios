60 /**
61 * @class LogWidget
62 * @brief Debug widget for displaying log output in a scrollable ImGui panel.
63 *
64 * This widget maintains separate buffers for each log scope and renders them
65 * in a scrollable text area. It supports filtering by log level, clearing
66 * the buffer, and auto-scrolling to the latest messages.
67 *
68 * Each scope has its own buffer with a maximum of 1000 entries, allowing
69 * efficient scope-based filtering without losing messages from other scopes.
70 *
71 * @note This widget uses internal locking for adding log entries from multiple threads.
72 */
73 class LogWidget : public ImGuiWidget {

183 /**
184 * @brief Adds an entry to a specific buffer, trimming if necessary.
185 */
186 void addToBuffer(std::vector<LogEntry>& buffer, LogEntry entry) {
187 buffer.push_back(std::move(entry));
188 if (maxEntries_ > 0 && buffer.size() > maxEntries_) {
189 const std::size_t overflow = buffer.size() - maxEntries_;
190 buffer.erase(buffer.begin(),
191 buffer.begin() + static_cast<std::ptrdiff_t>(overflow));

195 /**

198 [[nodiscard]] const std::vector<LogEntry>& activeBuffer() const {
199 if (activeScopeFilter_.empty()) {
200 auto it = scopeBuffers_.find(ALL_SCOPES_KEY);
201 if (it != scopeBuffers_.end()) {
202 return it->second;

206 if (it != scopeBuffers_.end()) {

248 /**
249 * @brief Generates a simple timestamp string (HH:MM:SS.mmm).
250 *
251 * @return The current time formatted as "HH:MM:SS.mmm".
252 */
253 [[nodiscard]] static std::string currentTimestamp() noexcept {
254 auto now = std::chrono::system_clock::now();
255 auto time_t_now = std::chrono::system_clock::to_time_t(now);
256 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
257 now.time_since_epoch()) % 1000;
258
259 std::tm tm_buf{};
260#ifdef _WIN32
261 localtime_s(&tm_buf, &time_t_now);
262#else
263 localtime_r(&time_t_now, &tm_buf);
264#endif
265 std::ostringstream oss;
266 oss << std::put_time(&tm_buf, "%H:%M:%S")
267 << '.' << std::setfill('0') << std::setw(3) << ms.count();
268 return oss.str();
269 }

275 /**
276 * @brief Adds a log entry to the appropriate scope buffer(s).
277 *
278 * Thread-safe. Adds to both the scope-specific buffer and the "all" buffer.
279 */
280 void addLog(LogLevel level, const std::string& scope, const std::string& message) {
281 // If user scrolls up and AutoScroll is off, we do not accept new entries
282 // to make sure memory is not flooded with new log entries in the background.
283 if (!acceptNewEntries_.load(std::memory_order_relaxed)) {
284 skippedEntries_.fetch_add(1, std::memory_order_relaxed);
285 return;
286 }
287
288 std::lock_guard<std::mutex> lock(bufferMutex_);
289
290 // Collect unique scopes for the filter dropdown
291 collectScope(scope);
292
293 LogEntry entry;
294 entry.level = level;
295 entry.scope = scope;
296 entry.message = message;
297 entry.timestamp = currentTimestamp();
298
299 // Add to scope-specific buffer
300 addToBuffer(scopeBuffers_[scope], entry);
301
302 // Add to "all scopes" buffer
303 addToBuffer(scopeBuffers_[ALL_SCOPES_KEY], entry);
304
305 // Always signal new content - scroll decision happens in draw()
306 scrollToBottom_ = true;
307 }

417 /**
418 * @brief Renders the log widget using ImGui.
419 */
420 void draw() override {
421 ImGui::SetNextWindowSize(ImVec2(600, 400), ImGuiCond_FirstUseEver);
422
423 // Window is dockable by default when ImGuiConfigFlags_DockingEnable is set
424 if (ImGui::Begin("Log Console", nullptr)) {
425
426 // Toolbar row
427 if (ImGui::Button("Clear")) {
428 clear();
429 }
430 ImGui::SameLine();
431
432 if (ImGui::Checkbox("Auto-Scroll", &autoScroll_)) {
433 // Detect transition from enabled to disabled
434 if (prevAutoScroll_ && !autoScroll_) {
435 scrollUpOneEntry_ = true;
436 }
437 }
438 prevAutoScroll_ = autoScroll_;
439 ImGui::SameLine();
440
441 // Level filter combo
442 const char* levelItems[] = { "Debug", "Info", "Warn", "Error" };
443 ImGui::SetNextItemWidth(80);
444 if (ImGui::Combo("Level", &filterLevelIndex_, levelItems, IM_ARRAYSIZE(levelItems))) {
445 filterLevel_ = static_cast<LogLevel>(filterLevelIndex_);
446 }
447 ImGui::SameLine();
448
449 // Scope filter combo
450 {
451 std::lock_guard<std::mutex> lock(bufferMutex_);
452
453 std::string scopePreview = (selectedScopeIndex_ == 0)
454 ? "All Scopes"
455 : (selectedScopeIndex_ <= static_cast<int>(collectedScopes_.size())
456 ? collectedScopes_[selectedScopeIndex_ - 1]
457 : "All Scopes");
458
459 ImGui::SetNextItemWidth(150);
460 if (ImGui::BeginCombo("Scope", scopePreview.c_str())) {
461 bool isSelected = (selectedScopeIndex_ == 0);
462 if (ImGui::Selectable("All Scopes", isSelected)) {
463 selectedScopeIndex_ = 0;
464 activeScopeFilter_.clear();
465 if (onScopeFilterChanged_) {
466 onScopeFilterChanged_("");
467 }
468 }
469 if (isSelected) {
470 ImGui::SetItemDefaultFocus();
471 }
472
473 for (std::size_t i = 0; i < collectedScopes_.size(); ++i) {
474 isSelected = (selectedScopeIndex_ == static_cast<int>(i + 1));
475 if (ImGui::Selectable(collectedScopes_[i].c_str(), isSelected)) {
476 selectedScopeIndex_ = static_cast<int>(i + 1);
477 activeScopeFilter_ = collectedScopes_[i];
478 if (onScopeFilterChanged_) {
479 onScopeFilterChanged_(activeScopeFilter_);
480 }
481 }
482 if (isSelected) {
483 ImGui::SetItemDefaultFocus();
484 }
485 }
486 ImGui::EndCombo();
487 }
488 }
489 ImGui::SameLine();
490
491 textFilter_.Draw("Filter", 150);
492
493 ImGui::Separator();
494
495 // Log area
496 const float footerHeight =
497 ImGui::GetStyle().ItemSpacing.y + ImGui::GetFrameHeightWithSpacing();
498 if (ImGui::BeginChild("LogScrollRegion", ImVec2(0, -footerHeight),
499 ImGuiChildFlags_Borders,
500 ImGuiWindowFlags_HorizontalScrollbar)) {
501
502 // Copy active buffer
503 std::vector<LogEntry> bufferCopy;
504 bool hasNewContent = false;
505 {
506 std::lock_guard<std::mutex> lock(bufferMutex_);
507 bufferCopy = activeBuffer();
508 hasNewContent = scrollToBottom_;
509 scrollToBottom_ = false;
510 }
511
512 ImGui::PushStyleVar(ImGuiStyleVar_ItemSpacing, ImVec2(0, 1));
513
514 for (const auto& entry : bufferCopy) {
515 // Filter by level
516 if (static_cast<int>(entry.level) < static_cast<int>(filterLevel_)) {
517 continue;
518 }
519
520 // Build display string for text filter
521 // If a specific scope is selected, omit scope from display (already visible in combo)
522 std::string displayLine;
523 if (activeScopeFilter_.empty()) {
524 // "All Scopes" selected - show scope in each line
525 displayLine = entry.timestamp + " " +
526 labelForLevel(entry.level) + " [" +
527 entry.scope + "] " + entry.message;
528 } else {
529 // Specific scope filtered - omit scope from display
530 displayLine = entry.timestamp + " " +
531 labelForLevel(entry.level) + " " + entry.message;
532 }
533
534 // Apply text filter
535 if (!textFilter_.PassFilter(displayLine.c_str())) {
536 continue;
537 }
538
539 // Render with color
540 ImVec4 color = colorForLevel(entry.level);
541 ImGui::PushStyleColor(ImGuiCol_Text, color);
542 ImGui::TextUnformatted(displayLine.c_str());
543 ImGui::PopStyleColor();
544 }
545
546 ImGui::PopStyleVar();
547
548 // Scroll state
549 float scrollY = ImGui::GetScrollY();
550 float scrollMaxY = ImGui::GetScrollMaxY();
551 bool atBottomNow = (scrollMaxY <= 0.0f) || (scrollY >= scrollMaxY - 5.0f);
552
553 // Nudge scroll up when auto-scroll was just disabled
554 // This prevents showing that new entries are being added
555 if (scrollUpOneEntry_) {
556 // Calculate offset for 2 log entries
557 float lineHeight = ImGui::GetTextLineHeightWithSpacing();
558 float nudgeAmount = lineHeight * 2.0f;
559
560 // Set scroll position relative to maximum (scroll up from bottom)
561 float targetScrollY = scrollMaxY - nudgeAmount;
562 if (targetScrollY < 0.0f) targetScrollY = 0.0f;
563 ImGui::SetScrollY(targetScrollY);
564 scrollUpOneEntry_ = false;
565
566 // Update atBottomNow since we just scrolled
567 atBottomNow = false;
568 }
569 // Auto-scroll decision
570 else if (hasNewContent && (autoScroll_ || wasAtBottom_)) {
571 ImGui::SetScrollHereY(1.0f);
572 }
573
574 // Pause logging if:
575 // - AutoScroll is off AND user has not scrolled to lower bottom
576 bool pauseLogging = (!autoScroll_ && !atBottomNow);
577 acceptNewEntries_.store(!pauseLogging, std::memory_order_relaxed);
578 wasAtBottom_ = atBottomNow;
579 }
580 ImGui::EndChild();
581
582 // Footer
583 {
584 std::lock_guard<std::mutex> lock(bufferMutex_);
585 ImGui::Text("Entries: %zu / %zu", activeBuffer().size(), maxEntries_);
586 }
587
588 bool loggingPaused = !acceptNewEntries_.load(std::memory_order_relaxed);
589 auto skipped = skippedEntries_.load(std::memory_order_relaxed);
590
591 if (loggingPaused) {
592 ImGui::SameLine();
593 if (skipped > 0) {
594 ImGui::TextColored(ImVec4(1.0f, 0.8f, 0.2f, 1.0f),
595 "Logging paused (%zu entries skipped)", skipped);
596 } else {
597 ImGui::TextColored(ImVec4(1.0f, 0.8f, 0.2f, 1.0f), "Logging paused");
598 }
599 } else if (skipped > 0) {
600 ImGui::SameLine();
601 ImGui::TextColored(ImVec4(0.6f, 0.8f, 1.0f, 1.0f),
602 "%zu entries were skipped", skipped);
603 skippedEntries_.store(0, std::memory_order_relaxed);
604 }
605 }
606 ImGui::End();
607 }
