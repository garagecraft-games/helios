24 /**
25 * @brief Central registry for managing game entities, systems, and the active level.
26 *
27 * @details
28 * The GameWorld is the root container for the game state. It manages the lifecycle
29 * of GameObjects, executes Systems, and holds the current Level.
30 *
31 * Key responsibilities:
32 * - **Entity Management:** Owns all GameObjects and provides lookup by Guid.
33 * - **System Execution:** Manages and updates registered Systems.
34 * - **Level Management:** Holds the active Level instance.
35 * - **Update Loop:** Orchestrates the frame update by updating GameObjects and then Systems.
36 */
37 class GameWorld {

41 * @brief Hash map storing all active GameObjects, indexed by their Guid.

43 * @details Uses std::unordered_map for O(1) average-case lookup performance.
44 * The map owns all GameObjects via std::unique_ptr.
45 */
46 std::unordered_map<helios::util::Guid, std::unique_ptr<GameObject>> gameObjects_;

106 /**
107 * @brief Adds a System to the GameWorld.
108 *
109 * @details Creates a System of the specified type with the given constructor arguments,
110 * invokes its `onAdd()` callback, and registers it with this GameWorld.
111 * Systems are updated each frame after GameObjects.
112 *
113 * @tparam T The System type to add. Must derive from System.
114 * @tparam Args Constructor argument types.
115 * @param args Arguments forwarded to the System constructor.
116 *
117 * @return Reference to the newly added System.
118 *
119 * @note The GameWorld takes ownership of the System.
120 */
121 template<typename T, typename... Args>
122 T& add(Args&&... args) {
123 auto system_ptr = std::make_unique<T>(std::forward<Args>(args)...);
124 T* raw_ptr = system_ptr.get();
125
126 system_ptr->onAdd(this);
127 systems_.push_back(std::move(system_ptr));
128
129 return *raw_ptr;
130 }
131
132 /**
133 * @brief Retrieves a System of the specified type.
134 *
135 * @details Searches through all registered Systems using dynamic_cast
136 * to find a matching type.
137 *
138 * @tparam T The System type to retrieve. Must derive from System.
139 *
140 * @return Pointer to the System if found, nullptr otherwise.
141 *
142 * @note Uses dynamic_cast internally; consider type-based registry for
143 * performance-critical paths.
144 */
145 template<typename T>
146 [[nodiscard]] T* get() const {
147 for (const auto& system : systems_) {
148 if (auto* sys = dynamic_cast<T*>(system.get())) {
149 return sys;
150 }
151 }
152 return nullptr;
153 }

155 /**

159 *
160 * @return True if a System of type T exists, false otherwise.
161 */
162 template<typename T>
163 [[nodiscard]] bool has() const {
