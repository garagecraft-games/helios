31 /**
32 * @brief Represents a hierarchical SceneGraph that organizes all renderable
33 * and non-renderable objects within a scene.
34 *
35 * Every scene has one implicit root node all nodes are appended to when
36 * `addNode` is called.
37 *
38 * A Scene is responsible for managing the SceneGraph, consisting of SceneNodes;
39 * it provides methods to frustum cull SceneNodes based on a specific `FrustumCullingStrategy`
40 * and a particular `CameraSceneNode` and also allows for propagating world transformations
41 * through the contained subtrees.
42 *
43 * @see [RTR, pp. 828]
44 * @see [Gre19, pp. 693]
45 * @see [HDMS+14, pp. 138]
46 *
47 * @note By default, we're not allowing copying/moving Scenes.
48 * These constraints may be relaxed in the future, depending on valid use cases.
49 */
50 class Scene {

66 /**
67 * @brief The root node of this scene.
68 *
69 * All other SceneNodes are (in)direct children of this node.
70 */
71 std::unique_ptr<SceneNode> root_;
72
73 /**
74 * @brief Internal helper function to force-propagate the worldTransformation of SceneNodes to their child nodes.
75 *
76 * Propagation is cancelled for this node if the resulting world transform
77 * is considered equal to the current world transform of the SceneNode.
78 *
79 * @param node The current SceneNode for which to propagate the world transformation.
80 * @param parentWorldTransform The world transformation from the parent node.
81 *
82 * @see updateNodes()
83 */
84 void propagateWorldTransform(SceneNode& node, const math::mat4f& parentWorldTransform) const;
85
86 /**
87 * @brief Selectively updates the world transformation of SceneNodes in the scene graph.
88 *
89 * A SceneNode is forced to update its own world transformation and the
90 * world transformation of its children as soon as a check for the dirty-state of the
91 * node is true.
92 *
93 * @param node The current SceneNode processed.
94 * @param wt The world transform to apply.
95 *
96 * @note It is important that this method is called before the next render pass, or before
97 * the scene graph is culled. This method must also consider CameraSceneNodes, making sure the
98 * associated `Camera` is updated with the view matrix.
99 */
100 void updateNodes(SceneNode& node, const math::mat4f& wt) const;
