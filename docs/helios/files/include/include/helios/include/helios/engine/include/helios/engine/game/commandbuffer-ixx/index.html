24 /**
25 * @brief Queue for deferred execution of game commands.
26 *
27 * @details The `CommandBuffer` provides a queue for game commands, decoupling their
28 * creation from their execution. Commands are buffered along with their target
29 * `GameObject` identifiers (`Guid`) and are executed in a single batch when `flush()`
30 * is called against a `GameWorld`.
31 *
32 * Typical usage pattern:
33 * ```cpp
34 * helios::engine::game::CommandBuffer cmdBuffer;
35 * helios::engine::game::GameWorld world;
36 *
37 * // Queue commands during input processing
38 * cmdBuffer.add(player.guid(), std::make_unique<MoveCommand>(direction, speed));
39 * cmdBuffer.add(enemy.guid(), std::make_unique<AttackCommand>(target));
40 *
41 * // Execute all commands at end of frame
42 * cmdBuffer.flush(world); // Executes and clears buffer
43 * ```
44 *
45 * @note Commands targeting non-existent GameObjects are silently skipped during flush().
46 * This behavior will be logged once the logging system is available.
47 */
48 class CommandBuffer {
49
50 /**
51 * @brief Internal struct pairing a command with its target GameObject identifier.
52 */
53 struct TargetedCommand {
54 helios::util::Guid guid;
55 std::unique_ptr<helios::engine::game::Command> command;
56 };
57
58 /**
59 * @brief Internal queue storing buffered commands.
60 *
61 * @details Uses std::deque for efficient push_back and iteration.
62 */
63 std::deque<TargetedCommand> commandBuffer_;

66 * @brief The logger used with this CommandBuffer instance.

68 * Defaults to HELIOS_LOG_SCOPE.

70 inline static const helios::util::log::Logger& logger_ = helios::util::log::LogManager::loggerForScope(
71 HELIOS_LOG_SCOPE);

75 /**
76 * @brief Adds a command targeting a GameObject identified by Guid.
77 *
78 * @param guid The unique identifier of the target GameObject.
79 * @param command The command to execute. Ownership is transferred to the buffer.
80 *
81 * @pre command must not be nullptr.
82 *
83 * @note If command is nullptr, the call is silently ignored (will be logged in future).
84 * @note This overload is useful when the GameObject instance is not directly available,
85 * such as in networked scenarios or when processing serialized commands.
86 */
87 void add(const helios::util::Guid& guid, std::unique_ptr<helios::engine::game::Command> command);
88
89 /**
90 * @brief Executes all buffered commands against the GameWorld and clears the buffer.
91 *
92 * @param gameWorld The world containing target GameObjects.
93 *
94 * @return Reference to this CommandBuffer for method chaining.
95 *
96 * @note Commands targeting non-existent GameObjects (i.e., find() returns nullptr)
97 * are skipped and a warning is logged.
98 * @note All commands are cleared after execution, regardless of success or failure.
99 *
100 * @warning If a Command::execute() throws an exception, the flush operation will abort
101 * and remaining commands will not be executed. Consider making execute() noexcept
102 * or handling exceptions within command implementations.
103 */
104 CommandBuffer& flush(helios::engine::game::GameWorld& gameWorld);
105
106 /**
107 * @brief Clears all buffered commands without executing them.
108 *
109 * @return Reference to this CommandBuffer for method chaining.
110 *
111 * @note This destroys all buffered commands. Use this to discard commands
112 * without executing them (e.g., when reverting to a previous game state).
113 */
114 CommandBuffer& clear();
115 };
