22 /**
23 * @brief Represents a 2-dimensional vector of the generic type <T>.
24 *
25 * The `vec2` struct provides a lightweight and efficient way to handle 2D
26 * vector mathematics for the numeric types float, int, double. For convenient access,
27 * the shorthands `vec2f`, `vec2d` and `vec2i` are available.
28 *
29 * @tparam T The numeric type of the vector components.
30 */
31 template<helios::math::Numeric T>
32 struct vec2 {

34 private:

36 * @brief Internal array storing the vector components.
37 */
38 T v[2];

40 public:

42 /**
43 * @brief Creates a new vec2 with its values initialized to (0, 0)
44 */
45 constexpr vec2() noexcept : v{static_cast<T>(0), static_cast<T>(0)} {}

49 * @brief Constructs a new vec2 with the specified x, y components.
50 *
51 * @param x The value for the x component.
52 * @param y The value for the y component.
53 */
54 constexpr vec2(const T x, const T y) noexcept : v{x, y} {}
55
56 /**
57 * @brief Converts the current vec2<T> into a 3D vector representation.
58 *
59 * @return A 3D vector corresponding to the converted representation of the object or input.
60 */
61 constexpr helios::math::vec3<T> toVec3() const {
62 return helios::math::vec3<T>{v[0], v[1], static_cast<T>(0)};
63 }
64
65 /**
66 * @brief Provides read only access to the vector components.
67 * Bounds checking is performed via `assert` in debug builds.
68 *
69 * @param i The index to query
70 *
71 * @return A const ref to the requested component.
72 */
73 constexpr const T& operator[](const size_t i) const noexcept {
74 assert(i <= 1 && "vec2 - Index out of bounds.");
75 return this->v[i];
76 }
77
78 /**
79 * @brief Provides read-write access to the vector components.
80 * Bounds checking is performed via `assert` in debug builds.
81 *
82 * @param i The index to query
83 *
84 * @return A ref to the requested component.
85 */
86 constexpr T& operator[](const size_t i) noexcept {
87 assert(i <= 1 && "vec2 - Index out of bounds.");
88 return this->v[i];

91 /**
92 * @brief Returns the magnitude of this vec2<T>.
93 *
94 * @return The magnitude of this vector as type FloatingPointType<T>.
95 */
96 inline FloatingPointType<T> length() const noexcept {
97 if (v[0] == 0 && v[1] == 0) {
98 return static_cast<FloatingPointType<T>>(0);
99 }
100 return static_cast<FloatingPointType<T>>(
101 std::sqrt(
102 static_cast<double>(v[0]) * static_cast<double>(v[0]) +
103 static_cast<double>(v[1]) * static_cast<double>(v[1])
104 )
105 );
106 }
107
108 /**
109 * @brief Normalizes this vec2<T>.
110 *
111 * @return The normalized FloatingPointType<T> vector.
112 */
113 inline vec2<FloatingPointType<T>> normalize() const noexcept {
114 if (v[0] == 0 && v[1] == 0) {
115 return vec2<FloatingPointType<T>>(
116 static_cast<FloatingPointType<T>>(0),
117 static_cast<FloatingPointType<T>>(0)
118 );
119 }
120 return vec2<FloatingPointType<T>>(
121 static_cast<FloatingPointType<T>>(v[0]) / length(),
122 static_cast<FloatingPointType<T>>(v[1]) / length()
123 );
124 }
125
126 /**
127 * @brief Strictly compares the elements of this vector with the elements
128 * of the rgt vector.
129 *
130 * @param rgt The right vector to compare for equal values
131 *
132 * @return True if all elements are equal (==), false otherwise.
133 */
134 constexpr bool operator==(const vec2<T>& rgt) const {
135 return v[0] == rgt[0] && v[1] == rgt[1];
136 }
137
138 /**
139 * @brief Compares this vector's elements with the rgt vector considering
140 * an epsilon value.
141 * Returns true if for all elements the equation |a-b| <= epsilon
142 * holds.
143 *
144 * @param rgt The other vector to compare with this vector for equality.
145 * @param epsilon The epsilon value to use for comparison. If omitted, the default epsilon (0.0001) is used.
146 *
147 * @return True if the elements of the vectors are considered equal,
148 * otherwise false.
149 *
150 * @see https://realtimecollisiondetection.net/blog/?p=89
151 *
152 * @todo account for abs (values close to zero) and rel
153 * (larger values), move epsilon to global constant?
154 */
155 constexpr bool same(const vec2<T>& rgt, T epsilon = 0.0001) const {
156 return std::fabs(v[0] - rgt[0]) <= epsilon &&
157 std::fabs(v[1] - rgt[1]) <= epsilon;
158 }
159
160 };

162 /**
163 * @brief Multiplies a 2D vector by a scalar value.
164 *
165 * @tparam T The numeric type of the vector components.
166 * @param v The vec2<T> vector to be multiplied.
167 * @param n The scalar vector to multiplay the vector by.
168 *
169 * @return a new vec2<T> instance representing the result of the scalar
170 * multiplication.
171 */
172 template<helios::math::Numeric T>
173 constexpr vec2<T> operator*(const vec2<T>& v, const T n) noexcept {
174 return vec2<T>{v[0] * n, v[1] * n};
175 }
176

178 /**
179 * @brief Computes the dot product of two 2D vectors.
180 *
181 * @tparam T The numeric type of the vector components.
182 * @param v1 The first vec2<T> vector.
183 * @param v2 The second vec2<T> vector.
184 *
185 * @return The dot product as a value of type T.
186 */
187 template<helios::math::Numeric T>
188 constexpr T dot(const vec2<T>& v1, const vec2<T>& v2) noexcept {
189 return v1[0]*v2[0] + v1[1]*v2[1];
190 }
