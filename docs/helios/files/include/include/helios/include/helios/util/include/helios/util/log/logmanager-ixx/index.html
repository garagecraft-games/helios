22 /**
23 * @brief LogManager for managing scoped Loggers and global sink configuration.
24 *
25 * The LogManager provides centralized control over logging output destinations.
26 * Sinks register themselves with a unique type identifier, allowing dynamic
27 * enable/disable without compile-time knowledge of all sink types.
28 *
29 * ```cpp
30 * // Register sinks (sinks define their own TYPE_ID)
31 * LogManager::getInstance().registerSink(std::make_shared<ConsoleSink>());
32 * LogManager::getInstance().registerSink(std::make_shared<ImGuiLogSink>(widget));
33 *
34 * // Enable/disable by type identifier
35 * LogManager::getInstance().enableSink("console");
36 * LogManager::getInstance().disableSink("imgui");
37 *
38 * // Check status
39 * if (LogManager::getInstance().isSinkEnabled("console")) { ... }
40 * ```
41 */
42 class LogManager {
