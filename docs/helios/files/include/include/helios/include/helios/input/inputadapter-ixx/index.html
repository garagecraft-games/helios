26 /**
27 * @brief Abstract interface for platform-specific input adapters.
28 *
29 * Provides a unified interface for querying input device states across different
30 * platforms. Concrete implementations translate generic input queries into
31 * platform-specific API calls.
32 *
33 * The adapter manages per-gamepad configuration through `GamepadSettings` and applies
34 * input normalization via a configurable `DeadzoneStrategy`.
35 *
36 * @see GLFWInputAdapter for a GLFW-based implementation.
37 * @see GamepadSettings for per-controller configuration options.
38 * @see DeadzoneStrategy for input normalization strategies.
39 */
40 class InputAdapter {
41
42 protected:
43 /**
44 * @brief Shared logger instance for all InputAdapter objects.
45 */
46 inline static const helios::util::log::Logger& logger_ = helios::util::log::LogManager::loggerForScope(HELIOS_LOG_SCOPE);
47
48 /**
49 * @brief Strategy used to normalize analog stick input within deadzones.
50 *
51 * Ownership is held by the adapter. Applied during gamepad state updates
52 * to filter out hardware drift and rescale input values.
53 */
54 std::unique_ptr<helios::input::gamepad::DeadzoneStrategy> deadzoneStrategy_;
55
56 /**
57 * @brief Per-gamepad configuration settings.
58 *
59 * Array indexed by gamepad ID, storing deadzone thresholds and axis
60 * inversion flags for each connected controller.
61 */
62 std::array<helios::input::gamepad::GamepadSettings, std::to_underlying(helios::input::types::Gamepad::size_)> gamepadSettings_ = {};

107 * @brief Updates the GamepadState objects with the values queried from the
108 * underlying Gamepad identified by the specified mask.
109 *
110 * Updates the GamepadState objects with the current values of the underlying
111 * hardware. If querying the underlying hardware fails, this method will
112 * update the GamepadState values to a valid initial state (no movement or interaction).
113 * Implementations should use `isConnected()` to check whether the current values of a
114 * `GamepadState` object can be trusted.
115 *
116 * @param gamepadMask A bitmask representing all GamepadState objects to update,
117 * e.g. Gamepad::ONE | Gamepad::TWO...
118 */
119 virtual void updateGamepadState(unsigned int gamepadMask) noexcept = 0;

122 /**
123 * @brief Returns a const ref to the GamepadState-object for the specified gamepadId.
124 *
125 * Implementing APIs can call this method in each frame to retrieve the current
126 * state for the gamepad, as computed by updateGamepadState().
127 * This method guarantees that for a given gamepadId, a single GamepadState object is
128 * reused across calls, avoiding re-instantiation overhead.
129 *
130 * @param gamepadId The id of the gamepad to query
131 *
132 * @return Returns a const ref to the GamepadState. If the GamepadState was never queried
133 * before, or if a recent call to updateGamepadState did not succeed, the values of the
134 * GamepadState will not hold representative values. Implementing APIs should verify the
135 * availability of the Gamepad by calling isConnected()
136 *
137 * @see updateGamepadState
138 * @see isConnected
139 */
140 [[nodiscard]] virtual const helios::input::gamepad::GamepadState& gamepadState(
141 helios::input::types::Gamepad gamepadId) const noexcept = 0;

144 /**
145 * @brief Returns the configuration settings for the specified gamepad.
146 *
147 * Provides access to the mutable settings object for the given gamepad,
148 * allowing runtime configuration of deadzone thresholds and axis inversion.
149 * Changes to the returned settings are applied during subsequent calls
150 * to `updateGamepadState()`.
151 *
152 * @param gamepadId The gamepad to retrieve settings for.
153 *
154 * @return Reference to the mutable GamepadSettings for the specified gamepad.
155 *
156 * @see GamepadSettings for available configuration options.
157 */

159 const auto id = static_cast<unsigned int>(gamepadId);
