189 /**
190 * @brief Compares this vector's elements with the rgt vector considering
191 * an epsilon value.
192 * Returns true if for all elements the equation |a-b| <= epsilon
193 * holds.
194 *
195 * @param rgt The other vector to compare with this vector for equality.
196 * @param epsilon The epsilon value to use for comparison. If omitted, the default epsilon (0.0001) is used.
197 *
198 * @return True if the elements of the vectors are considered equal,
199 * otherwise false.
200 *
201 * @see https://realtimecollisiondetection.net/blog/?p=89
202 *
203 * @todo account for abs (values close to zero) and rel
204 * (larger values), move epsilon to global constant?
205 */
206 constexpr bool same(const vec3<T>& rgt, T epsilon = 0.0001) const {
207 return std::fabs(v[0] - rgt[0]) <= epsilon &&
208 std::fabs(v[1] - rgt[1]) <= epsilon &&
209 std::fabs(v[2] - rgt[2]) <= epsilon;
210 }

272 * @param s The scalar divisor. Must not be zero.

334 * @tparam T The numeric type of the vector components.

403 template<helios::math::Numeric T>
404 inline vec3<FloatingPointType<T>> vec3<T>::normalize() const noexcept {
405 if (this->length() == static_cast<FloatingPointType<T>>(0)) {
406 return vec3<FloatingPointType<T>>(
407 static_cast<FloatingPointType<T>>(0),
408 static_cast<FloatingPointType<T>>(0),
409 static_cast<FloatingPointType<T>>(0)
410 );
411 }
412
413 return vec3<FloatingPointType<T>>(
414 static_cast<FloatingPointType<T>>(v[0]) / this->length(),
415 static_cast<FloatingPointType<T>>(v[1]) / this->length(),
416 static_cast<FloatingPointType<T>>(v[2]) / this->length()
417 );
418 }
