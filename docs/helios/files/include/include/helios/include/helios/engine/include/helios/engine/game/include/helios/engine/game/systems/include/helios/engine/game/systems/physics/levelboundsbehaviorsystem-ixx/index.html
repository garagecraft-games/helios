30 *
31 * @details
32 * This system checks if entities with movement components have exceeded the level bounds.
33 * When an entity leaves the bounds, it applies bounce behavior based on the
34 * LevelBoundsBehaviorComponent's restitution coefficient.
35 */

41 /**
42 * @brief Updates all entities that may have left level bounds.
43 *
44 * @details
45 * For each entity with the required components, checks if its world-space
46 * AABB is within level bounds. If not, applies bounce behavior.
47 *
48 * @param updateContext Context containing deltaTime and other frame data.
49 */
50 void update(helios::engine::game::UpdateContext& updateContext) noexcept override {
51
52 auto& gameObjects = gameWorld_->gameObjects();
53
54 for (auto& gameObjectPair : gameObjects) {
55
56 auto* obj = gameObjectPair.second.get();
57
58 auto* m2d = obj->get<helios::engine::game::components::physics::Move2DComponent>();
59
60 if (m2d) {
61
62 auto* ab = obj->get<helios::engine::game::components::model::ModelAabbComponent>();
63 auto* sc = obj->get<helios::engine::game::components::scene::SceneNodeComponent>();
64 auto* tc = obj->get<helios::engine::game::components::physics::TransformComponent>();
65 auto* bc = obj->get<helios::engine::game::components::physics::AabbColliderComponent>();
66 auto* bbc = obj->get<helios::engine::game::components::physics::LevelBoundsBehaviorComponent>();
67
68 if (ab && sc && tc && bc && bbc) {
69
70 auto objectBounds = bc->bounds();
71 auto levelBounds = gameWorld_->level().bounds();
72
73 if (!levelBounds.contains(objectBounds)) {
74
75 auto worldTranslation = tc->worldTranslation();
76 auto worldTransform = tc->worldTransform();
77
78 worldTranslation = bounce(worldTranslation, objectBounds, levelBounds, *m2d, *bbc);
79
80 auto& parentTransform = sc->sceneNode()->parent()->worldTransform();
81 auto parentTransform_inverse = parentTransform.inverse();
82
83 auto childLocalTranslation = parentTransform_inverse * worldTranslation.toVec4(1.0f);
84
85 tc->setLocalTranslation(childLocalTranslation.toVec3());
86 bc->setBounds(ab->aabb().applyTransform(
87 worldTransform.setTranslation(worldTranslation)
88 ));
89 }

99 /**
100 * @brief Applies bounce behavior to an entity that has left level bounds.
101 *
102 * @details
103 * Calculates the corrected position and reflected velocity when an entity
104 * collides with level boundaries. The restitution coefficient determines
105 * how much velocity is preserved after the bounce.
106 *
107 * @param worldTranslation Current world position of the entity.
108 * @param objectBounds World-space AABB of the entity.
109 * @param levelBounds World-space AABB of the level.
110 * @param m2d Reference to the Move2DComponent for velocity modification.
111 * @param bbc Reference to the LevelBoundsBehaviorComponent for restitution.
112 *
113 * @return Corrected world position after bounce.
114 */
115 helios::math::vec3f bounce(
116 helios::math::vec3f worldTranslation,
117 helios::math::aabbf objectBounds,
118 helios::math::aabbf levelBounds,
119 helios::engine::game::components::physics::Move2DComponent& m2d,
120 helios::engine::game::components::physics::LevelBoundsBehaviorComponent& bbc
121 ){
122
123
124 auto velocity = m2d.velocity();
125 bool hitX = false;
126 bool hitY = false;
127 auto translation = worldTranslation;
128
129 // Check left boundary collision
130 if (objectBounds.min()[0] < levelBounds.min()[0]) {
131 translation[0] += (levelBounds.min()[0] - objectBounds.min()[0]);
132 if (velocity[0] < 0) {
133 velocity = (velocity.withX(bbc.restitution() * -velocity[0]));
134 }
135 hitX = true;
136 // Check right boundary collision
137 } else if (objectBounds.max()[0] > levelBounds.max()[0]) {
138 translation[0] -= (objectBounds.max()[0] - levelBounds.max()[0]) ;
139 if (velocity[0] > 0) {
140 velocity = (velocity.withX(bbc.restitution() * -velocity[0]));
141 }
142 hitX = true;
143 }
144
145 objectBounds = objectBounds + (translation - worldTranslation);
146
147 // Check bottom boundary collision
148 if (objectBounds.min()[1] < levelBounds.min()[1]) {
149 translation[1] += (levelBounds.min()[1] - objectBounds.min()[1]);
150 if (velocity[1] < 0) {
151 velocity = velocity.withY(bbc.restitution() * -velocity[1]);
152 }
153 hitY = true;
154 // Check top boundary collision
155 } else if (objectBounds.max()[1] > levelBounds.max()[1]) {
156 translation[1] -= (objectBounds.max()[1] - levelBounds.max()[1]);
157 if (velocity[1] > 0) {
158 velocity = velocity.withY(bbc.restitution() * -velocity[1]);
159 }
160 hitY = true;
161 }
162
163 if (hitX || hitY) {
164 m2d.setVelocity(velocity);
165 }
166
167 return translation;
168 }
