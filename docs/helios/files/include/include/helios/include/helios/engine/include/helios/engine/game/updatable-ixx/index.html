15 /**
16 * @brief Interface for components that require per-frame updates.
17 *
18 * @details Components implementing this interface will have their update()
19 * method called each frame by the owning GameObject. This enables frame-dependent
20 * behaviors such as physics simulation, animation, and input processing.
21 *
22 * Example implementation:
23 * ```cpp
24 * class PhysicsComponent : public Component, public Updatable {
25 * public:
26 * void update(UpdateContext& ctx) noexcept override {
27 * velocity_ += gravity_ * ctx.deltaTime;
28 * position_ += velocity_ * ctx.deltaTime;
29 * }
30 * };
31 * ```
32 *
33 * @note Implementations must be noexcept to prevent exceptions from propagating
34 * during the game loop.
35 */
36 class Updatable {
37
38 public:
39 virtual ~Updatable() = default;
40 Updatable() = default;
41
42 /**
43 * @brief Called each frame to update the component state.
44 *
45 * @param updateContext Context containing deltaTime, input state, and references
46 * to CommandBuffer and GameWorld.
47 */
48 virtual void update(helios::engine::game::UpdateContext& updateContext) noexcept = 0;
49 };
